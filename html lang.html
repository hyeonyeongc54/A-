    function harvestVote(event, id) {
        if (isFinalized) return;
        const currentVotedId = localStorage.getItem(VOTED_KEY) ? 
Number(localStorage.getItem(VOTED_KEY)) : null;
        const targetOpinion = opinions.find(o => o.id === id);
        if (!targetOpinion) return;

            targetOpinion.votes = Math.max(0, targetOpinion.votes - 1);
            localStorage.removeItem(VOTED_KEY);
            saveToStorage();
            
            showModal('🍇 투표 취소', `"${targetOpinion.text}" 의견에 대한 투표가 취소
되었습니다. 다른 의견에 자유롭게 다시 투표하실 수 있습니다.`, () => {
                renderBoard();
            });
            return;
        }
        // 시나리오 2: 아직 아무 의견에도 투표하지 않은 상태 (클린한 첫 투표)
        if (currentVotedId === null) {
            castVote(event, id);
            return;
        }
        // 시나리오 3: 이미 다른 의견에 투표했는데, 새로운 의견으로 갈아타려고 할 때 
(변경 확인)
        const previousOpinion = opinions.find(o => o.id === currentVotedId);
        const previousText = previousOpinion ? `"${previousOpinion.text}"` : '이전 투
표';

    // 실제 투표 처리 보조 함수
    function castVote(event, id) {
        const targetOpinion = opinions.find(o => o.id === id);
        if (!targetOpinion) return;
        const oldVotes = targetOpinion.votes;
        targetOpinion.votes += 1;
        localStorage.setItem(VOTED_KEY, id);
        saveToStorage();
        // 파티클 좌표 획득 및 이펙트 소환
        const rect = event.target.getBoundingClientRect();
        createGrapeShower(rect.left + rect.width / 2, rect.top + rect.height / 2);
        // 해당 카드 바운스 애니메이션 적용
        const cardElement = document.getElementById(`card-${id}`);
        if (cardElement) {
            cardElement.classList.add('bounce');
            cardElement.addEventListener('animationend', () => {
                cardElement.classList.remove('bounce');
                renderBoard();
                // 투표로 새로 채택 기준(3개)을 딱 돌파했을 경우 고유 모달 알림
                if (oldVotes < PLEDGE_THRESHOLD && targetOpinion.votes >= 
PLEDGE_THRESHOLD) {
                    setTimeout(() => {
                        showModal(
                            '🎉 자치 규칙 승격 축하!',
                            `"${targetOpinion.text}" 의견이 투표수 
${PLEDGE_THRESHOLD}송이를 돌파하여 '마을 품질 서약 규칙'으로 정식 등극했습니
다!\n\n서약 규칙 확인 페이지로 지금 이동해 보시겠습니까?`,
                            function() {
                                switchTab('pledge');
                            },
                            function() {
                                // 그냥 이 페이지에 계속 머무르기
                            }
                        );
                    }, 250);
                }
            }, { once: true });
        } else {
            renderBoard();
        }
    }

            }
        }

        adoptedRules.sort((a, b) => b.votes - a.votes);

        });

        const listDiv = document.getElementById('resolutionsList');

            paper.classList.add('finalized');

            listDiv.style.display = 'flex';

        } else {
            paper.classList.remove('finalized');
            lockedBox.style.display = 'block';
            listDiv.style.display = 'none';
        }
    }

    // 포도 방울이 팡 터지는 이펙트
    function createGrapeShower(startX, startY) {
        const particles = ['🍇', '🔮', '🟢', '💜', '🍃', '💮', '📜'];
        const count = 16;
        for (let i = 0; i < count; i++) {
            const el = document.createElement('div');
            el.className = 'grape-particle';
            el.innerText = particles[Math.floor(Math.random() * particles.length)];
            
            const angle = Math.random() * Math.PI * 2;
            const distance = 80 + Math.random() * 100;
            const targetX = Math.cos(angle) * distance;
            const targetY = Math.sin(angle) * distance - 40;
            el.style.setProperty('--x', `${targetX}px`);
            el.style.setProperty('--y', `${targetY}px`);
            el.style.left = `${startX}px`;
            el.style.top = `${startY}px`;
            document.body.appendChild(el);
            el.addEventListener('animationend', () => {
                el.remove();
            });
        }
    }
